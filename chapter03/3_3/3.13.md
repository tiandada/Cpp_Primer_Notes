```
(a) vector<int> v1;
//v1是空的vector
(b) vector<int> v2(10);
//10个元素，每个都初始化为0
(c) vector<int> v3(10, 42);
//10个元素，每个都初始化为42
(d) vector<int> v4{10};
//1个元素，为10
(e) vector<int> v5{10, 42};
//2个元素，值为10， 42
(f) vector<string> v6{10};
//10个元素，默认初始化
(g) vector<string> v7{10, "hi"};
//10个元素，每个元素的值都是“hi”
```
