#include <iostream>
#include <iterator>
#include <algorithm>
#include <vector>

using std::cin;
using std::cout;
using std::endl;
using std::vector;
using std::istream_iterator;
using std::ostream_iterator;
using std::sort;
using std::copy;

int main()
{
        vector<int> vec;
        istream_iterator<int> in_iter(cin), eof;
        while(in_iter != eof)
                vec.push_back(*in_iter++);
        sort(vec.begin(), vec.end());
        copy(vec.cbegin(), vec.cend(), ostream_iterator<int>(cout, " "));
        cout << endl;
        return 0;
}
