class HasPtr {
public:
    HasPtr(const std::string &s = std::string()) : ps(new std::string(s)), i(0) { }
    HasPtr(const HasPtr &hp) : ps(new std::string(*hp.ps), i(hp.i)) {  }
    HasPtr& HasPtr::opertor=(const HasPtr &rhs)
    {
        std::string *tmp_ps = new std::string(*rhs.ps);
        delete ps;
        ps = tmp_ps;
        i = rhs.i;
        return *this;
    }
    ~HasPtr() {
        delete ps;
    }
private:

    int i;
};
