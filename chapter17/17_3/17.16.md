#include <iostream>
#include <string>
#include <regex>

int main()
{
        std::string pattern("[^c]ei");
        pattern = "[[:alpha:]]*" + pattern + "[[:alpha:]]*";
        std::regex r(pattern);
        r.assign("[^c]ei", std::regex::icase);
        std::smatch result;
        std::string test_str;
        while(std::cin >> test_str)
        {
                if(std::regex_search(test_str, result, r))
                {
                        std::cout << "matched word" << " " << result.str() << std::endl;
                }
        }
        return 0;
}
